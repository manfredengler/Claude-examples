Skills

Capacidades

Una Skill es una capacidad reutilizable que extiende lo que Claude puede hacer. Se declara en el contexto del sistema y Claude puede invocarla durante la conversación. En el entorno de GitHub Copilot, las skills aparecen en un bloque <available_skills>.

system_prompt.xml 
<!-- Bloque inyectado en el system prompt -->
<available_skills>
  <skill>
    <name>pdf-reader</name>
    <description>
      Extrae texto y estructura de archivos PDF.
      Úsala cuando el usuario suba o mencione un PDF.
    </description>
    <location>builtin</location>
  </skill>

  <skill>
    <name>code-reviewer</name>
    <description>
      Revisa código fuente en busca de bugs, problemas de
      seguridad y oportunidades de mejora. Soporta los
      lenguajes: Python, TypeScript, Go, Rust, Java.
    </description>
    <location>builtin</location>
  </skill>
</available_skills>
invocación (Claude interna) 
// Claude detecta que necesita la skill y la invoca:
const result = await invokeSkill('pdf-reader', {

skill_pdf_reader.py 
from anthropic import Anthropic
import base64, pathlib

client = Anthropic()

def pdf_skill(pdf_path: str, question: str) -> str:
    """Skill: leer un PDF y responder una pregunta sobre él."""
    data = base64.standard_b64encode(
        pathlib.Path(pdf_path).read_bytes()
    ).decode()

    msg = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=1024,
        messages=[{
            "role": "user",
            "content": [
                {
                    "type": "document",
                    "source": {
                        "type": "base64",
                        "media_type": "application/pdf",
                        "data": data,
                    },
                },
                {"type": "text", "text": question},
            ],
        }],
    )
    return msg.content[0].text

# Uso:
answer = pdf_skill("contrato.pdf", "¿Cuál es la fecha de vencimiento?")
print(answer)
💡 Las skills son estáticas: se declaran antes de la conversación. Los tools son dinámicos: Claude decide cuándo y cómo llamarlos en tiempo de ejecución.

Herramientas

tool_loop.py 
def run_with_tools(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    while True:
        response = client.messages.create(
            model="claude-opus-4-5",
            max_tokens=1024,
            tools=tools,
            messages=messages,
        )

        # Si Claude no necesita ningún tool, devolvemos la respuesta
        if response.stop_reason == "end_turn":
            return response.content[0].text

        # Procesar cada tool_use en el response
        tool_results = []
        for block in response.content:
            if block.type == "tool_use":
                result = execute_tool(block.name, block.input)
                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": json.dumps(result),
                })

        # Devolver los resultados a Claude y continuar
        messages.extend([
            {"role": "assistant", "content": response.content},
            {"role": "user",      "content": tool_results},
        ])


def execute_tool(name: str, inputs: dict) -> dict:
    if name == "get_weather":
        # En producción: llamada real a OpenWeatherMap / WeatherAPI
        return {"temp": 22, "condition": "soleado", "humidity": 45}
    if name == "search_web":
        return {"results": ["Resultado 1…", "Resultado 2…"]}
    raise ValueError(f"Tool desconocido: {name}")

answer = run_with_tools("¿Qué tiempo hace en Madrid ahora mismo?")
print(answer)

Commands

Comandos

Los Commands son instrucciones con barra (/comando) que Claude reconoce para ejecutar comportamientos predefinidos sin necesidad de una descripción en lenguaje natural. Se pueden registrar en la configuración del agente.

  - name: target

    required: true
---

Por favor, resume el siguiente contenido en exactamente 3 puntos clave,
cada uno en una línea comenzando con "•".

Contenido a resumir:
$ARGUMENTS
commands_api.py – registrar comandos via API 
import anthropic

client = anthropic.Anthropic()

# Comandos definidos como "prefilled prompts" en el system prompt
COMMANDS = {
    "/help":    "Lista todos los comandos disponibles y su uso.",
    "/clear":   "Reinicia la conversación y olvida el historial previo.",
    "/explain": "Explica el último bloque de código de forma detallada.",
    "/refactor":"Refactoriza el último bloque de código con mejores prácticas.",
}

SYSTEM = """Eres un asistente de programación.
Comandos disponibles:
""" + "\n".join(f"{k}: {v}" for k, v in COMMANDS.items())

def chat(history: list, user_input: str):
    history.append({"role": "user", "content": user_input})
    resp = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=512,
        system=SYSTEM,
        messages=history,
    )
    reply = resp.content[0].text
    history.append({"role": "assistant", "content": reply})
    return reply

history = []
print(chat(history, "/help"))
print(chat(history, "/explain\n```python\nfor i in range(10): print(i)\n```"))

Hooks

Eventos

Los Hooks son funciones que se disparan automáticamente en puntos específicos del ciclo de vida de Claude: antes de ejecutar un tool, después de recibir una respuesta, al iniciar/terminar una sesión, etc.

.claude/settings.json

        }]
      }
    ],
    "PostToolUse": [
      {
        "matcher": "*",
        "hooks": [{
          "type": "command",
          "command": "python hooks/audit_logger.py"
        }]
      }
    ],
    "Stop": [
      {
        "hooks": [{
          "type": "command",
          "command": "python hooks/session_summary.py"
        }]
      }
    ]
  }
}
hooks/pre_bash_validator.py 
import sys, json

"""
Hook PreToolUse: valida comandos bash antes de ejecutarlos.
Recibe JSON por stdin, escribe JSON por stdout.
Exit 0 = continuar | Exit 1 = bloquear
"""

BLOCKED_CMDS = ["rm -rf /", "dd if=/dev/zero", ":(){ :|:& };:"]

payload = json.loads(sys.stdin.read())
command  = payload.get("tool_input", {}).get("command", "")

for bad in BLOCKED_CMDS:
    if bad in command:
        print(json.dumps({
            "decision": "block",
            "reason": f"Comando peligroso detectado: '{bad}'"
        }))
        sys.exit(1)

print(json.dumps({"decision": "approve"}))
sys.exit(0)

import sys, json, datetime, pathlib

"""Hook PostToolUse: registra cada uso de herramienta en un log."""

LOG_FILE = pathlib.Path("logs/tool_audit.jsonl")
LOG_FILE.parent.mkdir(exist_ok=True)

payload = json.loads(sys.stdin.read())

entry = {
    "ts":       datetime.datetime.utcnow().isoformat(),
    "tool":     payload.get("tool_name"),
    "input":    payload.get("tool_input"),
    "exit_code":payload.get("tool_result", {}).get("exit_code"),
}

with LOG_FILE.open("a") as f:
    f.write(json.dumps(entry) + "\n")

sys.exit(0)

Ciclo de vida de una solicitud con hooks activos:

Usuario envía mensaje → Claude decide usar tool → 🪝 PreToolUse hook → Tool ejecutado → 🪝 PostToolUse hook → Claude responde → 🪝 Stop hook
⬆️

PreToolUse

Se dispara antes de ejecutar un tool. Puede bloquear la ejecución.

⬇️

PostToolUse

Se dispara después de que el tool retorna. Útil para auditoría.

▶️

PreTurn

Al inicio de cada turno del usuario. Inyecta contexto dinámico.

⏹️

Stop

Al finalizar la sesión. Genera resúmenes o limpia recursos.

Agents

Agentes

Un Agent es una instancia de Claude que opera en un bucle autónomo: percibe el entorno, razona, llama tools y genera acciones hasta completar un objetivo sin intervención humana en cada paso.

research_agent.py 
import anthropic, json

client = anthropic.Anthropic()

# Herramientas disponibles para el agente
AGENT_TOOLS = [
    {
        "name": "web_search",
        "description": "Busca en la web y devuelve resultados relevantes.",
        "input_schema": {
            "type": "object",
            "properties": {"query": {"type": "string"}},
            "required": ["query"],
        },
    },
    {
        "name": "save_notes",
        "description": "Guarda notas en un archivo local.",
        "input_schema": {
            "type": "object",
            "properties": {
                "filename": {"type": "string"},
                "content":  {"type": "string"},
            },
            "required": ["filename", "content"],
        },
    },
]


class ResearchAgent:
    """Agente que investiga un tema y guarda un informe."""

    def __init__(self, topic: str):
        self.topic   = topic
        self.history = []
        self.system  = (
            "Eres un agente investigador autónomo. "
            "Usa web_search para reunir información y "
            "save_notes para guardar el informe final."
        )

    def run(self) -> str:
        self.history.append({
            "role": "user",
            "content": f"Investiga: {self.topic}. Escribe un informe completo.",
        })

        while True:
            resp = client.messages.create(
                model="claude-opus-4-5",
                max_tokens=2048,
                system=self.system,
                tools=AGENT_TOOLS,
                messages=self.history,
            )

            if resp.stop_reason == "end_turn":
                return resp.content[0].text  # Tarea completada

            # Ejecutar tools y continuar
            results = []
            for block in resp.content:
                if block.type == "tool_use":
                    out = self._exec_tool(block.name, block.input)
                    results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": json.dumps(out),
                    })

            self.history.extend([
                {"role": "assistant", "content": resp.content},
                {"role": "user",      "content": results},
            ])

    def _exec_tool(self, name, inputs):
        if name == "web_search":
            return {"results": [f"Artículo sobre {inputs['query']}…"]}
        if name == "save_notes":
            with open(inputs["filename"], "w") as f:
                f.write(inputs["content"])
            return {"saved": inputs["filename"]}

# Uso:
agent = ResearchAgent("Historia del procesador ARM")
report = agent.run()
print(report)

Sub-Agents

Orquestación

Un Sub-Agent es un agente especializado lanzado por un orquestador (agente padre). El orquestador descompone el trabajo en tareas, delega cada una a sub-agentes y consolida los resultados.

🤖 Orchestrator → delega → 🔍 Research Sub-Agent + ✍️ Writer Sub-Agent + ✅ QA Sub-Agent → consolida → Resultado final
orchestrator.py 
import anthropic, concurrent.futures

client = anthropic.Anthropic()

def run_subagent(role: str, task: str) -> str:
    """Lanza un sub-agente con un rol y tarea específicos."""
    resp = client.messages.create(
        model="claude-haiku-4-5",  # sub-agentes usan modelo rápido
        max_tokens=1024,
        system=f"Eres un {role}. Completa la tarea de forma concisa.",
        messages=[{"role": "user", "content": task}],
    )
    return resp.content[0].text


def orchestrate(topic: str) -> str:
    """Orquestador que paraleliza tres sub-agentes."""
    tasks = {
        "investigador": f"Lista 5 hechos clave sobre: {topic}",
        "redactor":     f"Escribe una intro de 3 párrafos sobre: {topic}",
        "revisor QA":   f"Lista 3 posibles críticas al tema: {topic}",
    }

    # Ejecutar sub-agentes en paralelo
    results = {}
    with concurrent.futures.ThreadPoolExecutor() as pool:
        futures = {
            pool.submit(run_subagent, role, task): role
            for role, task in tasks.items()
        }
        for future in concurrent.futures.as_completed(futures):
            results[futures[future]] = future.result()

    # Agente orquestador consolida todo
    synthesis_prompt = f"""
    Combina los siguientes aportes en un artículo coherente:

    HECHOS:
    {results['investigador']}

    INTRODUCCIÓN:
    {results['redactor']}

    CRÍTICAS:
    {results['revisor QA']}
    """

    final = client.messages.create(
        model="claude-opus-4-5",  # orquestador usa modelo potente
        max_tokens=2048,
        messages=[{"role": "user", "content": synthesis_prompt}],
    )
    return final.content[0].text

article = orchestrate("inteligencia artificial en medicina")
print(article)
⚡ Los sub-agentes pueden ejecutarse en paralelo (ThreadPoolExecutor) para reducir la latencia total. El orquestador usa el modelo más capaz; los sub-agentes suelen usar un modelo más rápido y económico.

Harness

Testing

Un Harness es el andamiaje de pruebas que ejecuta agentes o flujos de Claude de forma controlada, verifica que las salidas cumplan criterios y mide métricas (precisión, coste, latencia).

test_harness.py 
import pytest, anthropic, time

client = anthropic.Anthropic()

# Casos de prueba: (entrada, criterio de aceptación)
TEST_CASES = [
    {
        "input":    "¿Cuánto es 2 + 2?",
        "expected": "4",
        "check":    lambda r: "4" in r,
    },
    {
        "input":    "Escribe 'Hola mundo' en Python",
        "expected": "print",
        "check":    lambda r: "print" in r.lower(),
    },
    {
        "input":    "Qué es un pointer en C?",
        "expected": "dirección de memoria",
        "check":    lambda r: "memor" in r.lower(),
    },
]

class HarnessResult:
    def __init__(self, case, response, passed, latency_ms):
        self.case       = case
        self.response   = response
        self.passed     = passed
        self.latency_ms = latency_ms

def run_harness(cases=TEST_CASES) -> list[HarnessResult]:
    results = []
    for case in cases:
        t0 = time.perf_counter()
        resp = client.messages.create(
            model="claude-haiku-4-5",
            max_tokens=256,
            messages=[{"role": "user", "content": case["input"]}],
        )
        latency = (time.perf_counter() - t0) * 1000
        text = resp.content[0].text
        results.append(HarnessResult(case, text, case["check"](text), latency))
    return results

# Tests pytest
@pytest.mark.parametrize("case", TEST_CASES)
def test_claude_response(case):
    resp = client.messages.create(
        model="claude-haiku-4-5", max_tokens=256,
        messages=[{"role": "user", "content": case["input"]}],
    )
    text = resp.content[0].text
    assert case["check"](text), f"Falló: {case['input']!r}\nRespuesta: {text}"
llm_judge_eval.py 
from anthropic import Anthropic

client = Anthropic()

JUDGE_PROMPT = """\
Eres un juez objetivo que evalúa respuestas de IA.
Puntúa de 1 a 5 según:
  5 = perfectamente correcto y completo
  3 = parcialmente correcto
  1 = incorrecto o irrelevante

Pregunta: {question}
Respuesta a evaluar: {answer}
Respuesta esperada de referencia: {reference}

Devuelve ÚNICAMENTE un JSON: {{"score": N, "reason": "..."}}
"""

def judge_response(question: str, answer: str, reference: str) -> dict:
    resp = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=256,
        messages=[{
            "role": "user",
            "content": JUDGE_PROMPT.format(
                question=question, answer=answer, reference=reference
            ),
        }],
    )
    import json, re
    text = resp.content[0].text
    m = re.search(r'\{.*\}', text, re.DOTALL)
    return json.loads(m.group()) if m else {"score": 0, "reason": "parse error"}

# Ejemplo de evaluación
result = judge_response(
    question="¿Cuál es la capital de Francia?",
    answer="La capital de Francia es París.",
    reference="París",
)
print(result)  # {"score": 5, "reason": "Respuesta correcta y directa."}

MCP – Model Context Protocol

Protocolo

El Model Context Protocol (MCP) es un estándar abierto de Anthropic para conectar modelos de IA con fuentes de datos y herramientas externas. Un MCP server expone recursos, tools y prompts que Claude puede consumir.

mcp_server.py 
from mcp.server.fastmcp import FastMCP
import httpx, datetime

# Crear el servidor MCP
mcp = FastMCP("mi-servidor")


# Tool: búsqueda en base de datos
@mcp.tool()
async def buscar_producto(nombre: str, max_precio: float = None) -> dict:
    """Busca productos en el catálogo."""
    # Simulación: en producción sería una DB real
    productos = [
        {"id": 1, "nombre": "Laptop Pro", "precio": 1299.0},
        {"id": 2, "nombre": "Mouse inalámbrico", "precio": 45.0},
    ]
    results = [p for p in productos
               if nombre.lower() in p["nombre"].lower()
               and (max_precio is None or p["precio"] <= max_precio)]
    return {"count": len(results), "products": results}


# Resource: lectura de archivos de configuración
@mcp.resource("config://app-settings")
def get_app_settings() -> str:
    """Expone la configuración de la app como recurso MCP."""
    return """{
  "version": "2.1.0",
  "features": ["search", "notifications", "export"],
  "max_upload_mb": 50
}"""


# Prompt: plantilla de informe
@mcp.prompt()
def informe_ventas(mes: str, año: int) -> str:
    """Genera un prompt para crear el informe de ventas mensual."""
    return f"""Analiza los datos de ventas de {mes}/{año}.
    Incluye: resumen ejecutivo, top-10 productos, tendencias y recomendaciones."""


if __name__ == "__main__":
    mcp.run()  # Inicia el servidor MCP (stdio o HTTP)
mcp_client.py 
import asyncio
from anthropic import Anthropic
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

client = Anthropic()

async def chat_with_mcp(user_message: str):
    server_params = StdioServerParameters(
        command="python", args=["mcp_server.py"]
    )

    async with stdio_client(server_params) as (read, write):
        async with ClientSession(read, write) as session:
            await session.initialize()

            # Obtener tools del servidor MCP
            tools_response = await session.list_tools()
            tools = [
                {
                    "name": t.name,
                    "description": t.description,
                    "input_schema": t.inputSchema,
                }
                for t in tools_response.tools
            ]

            messages = [{"role": "user", "content": user_message}]

            # Bucle tool-use estándar
            while True:
                resp = client.messages.create(
                    model="claude-opus-4-5", max_tokens=1024,
                    tools=tools, messages=messages,
                )
                if resp.stop_reason == "end_turn":
                    return resp.content[0].text
                results = []
                for block in resp.content:
                    if block.type == "tool_use":
                        out = await session.call_tool(block.name, block.input)
                        results.append({
                            "type": "tool_result",
                            "tool_use_id": block.id,
                            "content": str(out),
                        })
                messages.extend([
                    {"role": "assistant", "content": resp.content},
                    {"role": "user",      "content": results},
                ])

asyncio.run(chat_with_mcp("¿Tienes laptops por menos de 1500€?"))
